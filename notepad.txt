    <script>
        const notepad = document.getElementById('notepad');
        const charCount = document.getElementById('charCount');
        const wordCount = document.getElementById('wordCount');
        const lineCount = document.getElementById('lineCount');
        const lastSaved = document.getElementById('lastSaved');
        const fileInput = document.getElementById('fileInput');
        const filenameInput = document.getElementById('filenameInput');

        // Update statistik saat mengetik
        notepad.addEventListener('input', updateStats);

        function updateStats() {
            const text = notepad.value;
            
            // Hitung karakter
            charCount.textContent = text.length;
            
            // Hitung kata
            const words = text.trim().split(/\s+/).filter(word => word.length > 0);
            wordCount.textContent = words.length;
            
            // Hitung baris
            lineCount.textContent = text.split('\n').length;
        }

        // Fungsi untuk menyimpan file
        function saveFile() {
            const text = notepad.value;
            const filename = filenameInput.value.trim() || 'catatan.txt';
            
            const blob = new Blob([text], { type: 'text/plain' });
            const url = URL.createObjectURL(blob);
            
            const a = document.createElement('a');
            a.href = url;
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            
            // Update status
            const now = new Date();
            lastSaved.textContent = `Terakhir disimpan: ${now.toLocaleString('id-ID')}`;
        }

        // Fungsi untuk membuka file
        function openFile() {
            fileInput.click();
        }

        fileInput.addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (file && file.type === 'text/plain') {
                const reader = new FileReader();
                reader.onload = function(e) {
                    notepad.value = e.target.result;
                    updateStats();
                    
                    // Update nama file
                    filenameInput.value = file.name;
                    lastSaved.textContent = `File "${file.name}" berhasil dibuka`;
                };
                reader.readAsText(file);
            } else {
                alert('Silakan pilih file .txt');
            }
        });

        // Fungsi untuk menghapus semua teks
        function clearText() {
            if (notepad.value.length > 0) {
                if (confirm('Yakin ingin menghapus semua teks?')) {
                    notepad.value = '';
                    updateStats();
                    lastSaved.textContent = 'Teks telah dihapus';
                }
            }
        }

        // Fungsi untuk copy teks
        function copyText() {
            if (notepad.value.length > 0) {
                notepad.select();
                document.execCommand('copy');
                lastSaved.textContent = 'Teks berhasil disalin ke clipboard';
                
                // Hilangkan seleksi setelah 1 detik
                setTimeout(() => {
                    window.getSelection().removeAllRanges();
                }, 1000);
            } else {
                alert('Tidak ada teks untuk disalin');
            }
        }

        // Shortcut keyboard
        notepad.addEventListener('keydown', function(e) {
            // Ctrl+S untuk simpan
            if (e.ctrlKey && e.key === 's') {
                e.preventDefault();
                saveFile();
            }
            
            // Ctrl+O untuk buka
            if (e.ctrlKey && e.key === 'o') {
                e.preventDefault();
                openFile();
            }
        });

        // Inisialisasi stats
        updateStats();

        // Auto-save ke localStorage setiap 30 detik
        setInterval(() => {
            if (notepad.value.trim()) {
                const autoSaveData = {
                    text: notepad.value,
                    timestamp: new Date().toISOString()
                };
                // Note: localStorage tidak tersedia di Claude.ai, jadi kita skip ini
                console.log('Auto-save attempted:', autoSaveData);
            }
        }, 30000);
    </script>
